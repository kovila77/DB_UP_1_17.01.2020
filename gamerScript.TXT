(

create table gamer
(
	id integer not null
		primary key autoincrement,
	nickname text not null
		unique,
	server_id integer not null
		references server,
	coins integer default 0 not null
);

create table object
(
	id integer not null
		primary key autoincrement,
	name text not null,
	owner_gamer_id integer not null
		references gamer
);
